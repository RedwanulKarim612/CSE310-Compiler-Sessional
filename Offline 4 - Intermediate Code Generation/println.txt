PUSH BX
PUSH CX
PUSH DX

PUSH AX
MOV DL, '-'
MOV AH, 2
INT 21H

NEG AX

MOV CX, 0
MOV BX, 10D

JNE WHILE_2
MOV AH, 2
WHILE_3:
POP DX

INT 21H

POP DX
